import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
df= pd.read_csv("amazon_ecommerce_1M.csv")
df.head()
df.info()
df.describe()
df.isnull().sum()
df.drop_duplicates(inplace=True)
df.columns
df["purchase_date"]= pd.to_datetime(df["purchase_date"])
df["year"]=df["purchase_date"].dt.year
df["month"]= df["purchase_date"].dt.month_name()
df.drop(columns=["rating","review_count","seller_rating","shipping_time_days","device","is_returned","delivery_status"], inplace=True)
df.groupby("category")["final_price"].sum()
df.groupby("location")["final_price"].sum().sort_values(ascending=False)
df[df["final_price"]<0]
sns.barplot(x="category", y="final_price", data=df)
plt.show()
sns.heatmap(df.corr(numeric_only=True),annot=True)
plt.show()
